--------------------------------------------------------
--  DDL for Table MANAGEMENT_LOG
--------------------------------------------------------

CREATE TABLE "MANAGEMENT" 
   (	"FACILITY_ID" NUMBER(*,0), 
	"DATE" DATE, 
	"MANAGER" VARCHAR2(100 BYTE)
   ) ;

--------------------------------------------------------
--  DDL for Table FACILITY
--------------------------------------------------------

CREATE TABLE "FACILITY" 
   (	"FACILITY_ID" NUMBER(*,0), 
	"FACILITY_NAME" VARCHAR2(100 BYTE), 
	"CAPACITY" NUMBER(*,0), 
	"LOCATION" VARCHAR2(255 BYTE), 
	"USAGE_GUIDELINES" CLOB, 
	"AVAILABLE_START_TIME" NUMBER(*,0), 
	"AVAILABLE_END_TIME" NUMBER(*,0), 
	"DEPARTMENT_ID" NUMBER(*,0)
   ) ;

--------------------------------------------------------
--  DDL for Table REPORT_HISTORY
--------------------------------------------------------

CREATE TABLE "REPORT" 
   (	"RESERVATION_ID" VARCHAR2(25 BYTE), 
	"REPORT_ID" NUMBER(*,0), 
	"REPORT_CONTENT" CLOB, 
	"PROCESSING_STATUS" VARCHAR2(100 BYTE)
   ) ;

--------------------------------------------------------
--  DDL for Table RESERVATION_HISTORY
--------------------------------------------------------

CREATE TABLE "RESERVATION" 
   (	"STUDENT_ID" NUMBER(*,0), 
	"FACILITY_ID" NUMBER(*,0), 
	"RESERVATION_ID" VARCHAR2(25 BYTE), 
	"RESERVATION_DATE" DATE, 
	"START_TIME" NUMBER(2,0),
    "END_TIME" NUMBER(2,0)
   ) ;

--------------------------------------------------------
--  DDL for Table DEPARTMENT
--------------------------------------------------------

CREATE TABLE "DEPARTMENT" 
   (	"DEPARTMENT_ID" NUMBER(*,0), 
	"DEPARTMENT_NAME" VARCHAR2(100 BYTE), 
	"PHONE_NUMBER" VARCHAR2(20 BYTE), 
	"OFFICE" VARCHAR2(100 BYTE)
   ) ;

--------------------------------------------------------
--  DDL for Table STUDENT
--------------------------------------------------------

CREATE TABLE "STUDENT" 
   (	"STUDENT_ID" NUMBER(*,0), 
	"NAME" VARCHAR2(100 BYTE), 
	"GENDER" CHAR(1 BYTE),
